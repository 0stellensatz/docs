\makeatletter
\def\cleartheorem#1{%
    \expandafter\let\csname#1\endcsname\relax
    \expandafter\let\csname c@#1\endcsname\relax
}
\makeatother

\theoremstyle{latinplain}

\newtheorem{theorem}{\termTheorem}[section]
\crefname{theorem}{\MakeLowercase{\termTheorem}}{\MakeLowercase{\termTheorems}}
\newtheorem{proposition}[theorem]{\termProposition}
\crefname{proposition}{\MakeLowercase{\termProposition}}{\MakeLowercase{\termPropositions}}
\newtheorem{lemma}[theorem]{\termLemma}
\crefname{lemma}{\MakeLowercase{\termLemma}}{\MakeLowercase{\termLemmas}}
\newtheorem{conjecture}[theorem]{\termConjecture}
\crefname{conjecture}{\MakeLowercase{\termConjecture}}{\MakeLowercase{\termConjectures}}
\newtheorem{corollary}[theorem]{\termCorollary}
\crefname{corollary}{\MakeLowercase{\termCorollary}}{\MakeLowercase{\termCorollaries}}
\newtheorem{problem}[theorem]{\termProblem}
\crefname{problem}{\MakeLowercase{\termProblem}}{\MakeLowercase{\termProblems}}

\theoremstyle{definition}

\newtheorem{definition}[theorem]{\termDefinition}
\crefname{definition}{\MakeLowercase{\termDefinition}}{\MakeLowercase{\termDefinitions}}
\newtheorem{example}[theorem]{\termExample}
\crefname{example}{\MakeLowercase{\termExample}}{\MakeLowercase{\termExamples}}

\newtheorem{numbered}[theorem]{\ignorespaces}

\theoremstyle{remark}

\newtheorem*{note}{\termNote}
\newtheorem*{remark}{\termRemark}
\newtheorem*{solution}{\termSolution}
